## 2. 基本 DGP：500 单位 × 10 期面板

### 原代码（有 Bug）

```stata
program define gen_panel
    syntax, [n(integer 500) t(integer 10) rho(real 0.8) te(real 0.0) seed(integer 0)]
    clear
    set seed `seed'
    set obs `n'
    gen unit = _n
    gen unit_fe = rnormal(0, 1)
    
    * 随机选 1/5 的单位作为处理组
    gen u = runiform()
    sort u
    gen treated = (_n <= `n' / 5)
    drop u
    
    expand `t'
    bysort unit: gen t = _n
    
    * 时间效应 ← 问题 2：只有 t=1 有值，t>=2 为缺失
    gen time_fe = rnormal(0, 0.5) if t == 1
    bysort t: egen tfe = mean(time_fe)
    drop time_fe
    rename tfe time_fe
    
    * AR(1) 误差 ← 问题 1：重复/冲突代码
    sort unit t
    gen eps = rnormal(0, 1) if t == 1
    replace eps = `rho' * L.eps + rnormal(0, 1) if t > 1 & !missing(L.eps)  // ← 此时未 xtset，L.eps 会报错
    xtset unit t
    replace eps = `rho' * eps[_n-1] + rnormal(0, 1) if t > 1 & unit == unit[_n-1]  // ← 覆盖上一行
    
    * 处理指示与结果
    gen D = (treated == 1 & t >= 5)
    gen Y = unit_fe + time_fe + `te' * D + eps
end
```

### 改正后的代码

```stata
program define gen_panel
    syntax, [n(integer 500) t(integer 10) rho(real 0.8) te(real 0.0) seed(integer 20260516)]
    clear
    set seed `seed'
    set obs `n'
    gen unit = _n
    gen unit_fe = rnormal(0, 1)
    
    * 随机选 1/5 的单位作为处理组
    gen u = runiform()
    sort u
    gen treated = (_n <= ceil(`n' / 5))
    drop u
    
    expand `t'
    bysort unit: gen t = _n
    
    * 时间效应：每个时期独立生成 ← 修复问题 2
    gen time_fe = .
    forvalues tt = 1/`t' {
        replace time_fe = rnormal(0, 0.5) if t == `tt'
    }
    
    * AR(1) 误差 ← 修复问题 1：先 xtset，再生成 eps
    sort unit t
    xtset unit t
    gen eps = rnormal(0, 1) if t == 1
    replace eps = `rho' * eps[_n-1] + rnormal(0, 1) if t > 1 & unit == unit[_n-1]
    
    * 处理指示与结果
    gen D = (treated == 1 & t >= 5)
    gen Y = unit_fe + time_fe + `te' * D + eps
end
```

## 3. 单次运行：观察三种标准误（无需修改）

```stata
gen_panel, n(500) t(10) rho(0.8) te(0.0) seed(42)

* iid 标准误
```

```stata
local nreps = 300
forvalues r = 1/`nreps' {
```
